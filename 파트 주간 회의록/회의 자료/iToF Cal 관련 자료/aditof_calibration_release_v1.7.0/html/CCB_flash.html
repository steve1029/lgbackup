import numpy as np
import os
import sys
import yaml

# sys.path.append("./aditofdevicepython/")
sys.path.insert(1,"./aditofdevicepython/")
# abspath = os.path.abspath(__file__)
# dname = os.path.dirname(abspath)
# os.chdir(dname)
import aditofpython as tof
# os.chdir(os.path.dirname(dname))


import shutil

def ProgramModuleFlash(config_file=None, CalibfolderName=None):
        '''
                Flashes compressed ccb into module 

                Inputs: 
                  config_file           - name of configuration (.cfg) file 
                  CalibfolderName       - name of calbration folder 
                  
                Outputs: 
                  None 
                
        '''

        pythonEXEPath = os.path.dirname(sys.executable)
        print('pythonEXEPath:',pythonEXEPath)
        shutil.copyfile(os.path.join("aditofdevicepython", "tofi_processor.obj"), os.path.join(pythonEXEPath, "tofi_processor.obj"))

        with open(os.path.join(CalibfolderName, 'header.yaml')) as y:
                header = yaml.load(y, Loader=yaml.SafeLoader)

        flashJsonPath = header['flashConfigJson']
        if flashJsonPath == '':
                print("\nGenerate compressed ccb first!!\n")
                return False

        CompCCBPath = header['compressed_ccb']
        if CompCCBPath == '':
                print("\nGenerate compressed ccb first!!\n")
                return False

        CFGPath = header['sensor_firmware']
        if CFGPath == '':
                print("\nGenerate compressed ccb first!!\n")
                return False

        system = tof.System()
        status = system.initialize()
        print("system.initialize()", status)

        cameras = []
        status = system.getCameraList(cameras)
        print("system.getCameraList()", status)

        print("Cameras: ", cameras)
        if len(cameras) < 1:
                print("No cameras detected")
                return False;

        camera1 = cameras[0]
        print("\nFLASHING: " + str(flashJsonPath) + '\n')
        status = camera1.initialize(flashJsonPath)
        print("camera1.initialize()", status)
        assert(status == tof.Status.Ok)

        status = camera1.powerUp()
        print("camera1.powerUp()", status)
        assert(status == tof.Status.Ok)

        status = camera1.stop()
        print("camera1.stop()", status)
        assert(status == tof.Status.Ok)

        status = camera1.writeModuleEeprom(CompCCBPath, CFGPath);
        print("camera1.writeModuleEeprom()", status)
        if (status == tof.Status.Ok):
                print("Flash programming complete")
                header['compressed_ccb_flashed'] = True
                with open(os.path.join(CalibfolderName,'header.yaml'),'w') as f:
                        yaml.dump(header, f)
                print("header.yaml updated")
                print("\n=================")
                print("FLASH SUCCESSFUL!")
                print("=================")
        else:
                print("Flash programming failed")
                return False

        return True
